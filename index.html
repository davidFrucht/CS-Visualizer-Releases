static int FindMax(int[] arr)
{
    int max = arr[0];
    for (int i = 1; i < arr.Length; i++)
    {
        if (arr[i] > max)
        {
            max = arr[i];
        }
    }
    return max;
}

{
    public string Name;
    public virtual void Speak() { }
}

class Dog : Animal
{
    public override void Speak()
        => Console.Write("Woof!");
}

Animal a = new Dog();
a.Name = "Buddy";
a.Speak(); // virtual → Dog
